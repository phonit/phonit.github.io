centos7 修复启动

ls
grub>insmod xfs
grub>set root=(hd0,1)
grub>linux16 /vmlinuz-xxxxx root=/dev/mapper/centos-root #此处xxxxxx指的是本机tab键自动补全的文件 root=填写本机根目录挂载的分区
grub>initrd16 /initramfs-.xxxxx.img
grub>boot
mv /boot/grub2/grub.cfg.bak /boot/grub2/grub.cfg

 

grub安装

grub2-install --boot-directory=/dev/sdb
grub2-install /dev/sdb --force 

grub2-mkconfig -o /etc/grub2/grub.cfg
