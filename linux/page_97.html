tar xvf mysql-boost-5.7.30.tar.gz
 
mkdir /app /data/mysql/3306  -p
 
ln -s /app/mysql-5.7.30   /usr/local/mysql
 
cd /usr/local/mysql
mkdir bld
cd bld
 
yum install cmake  gcc gcc-c++ openssl-devel  ncurses-devel -y
 
cmake .. -DWITH_BOOST=../boost/
 
make  && make install 【DESTDIR=/app/mysql】
 
useradd -r -s /sbin/nologin  mysql
chown -R mysql. /data/
 
 
vi /etc/profile.d/mysql.sh
 
PATH=/usr/local/mysql/bin:$PATH
 
 
5.7初始化
mysqld --initialize-insecure --user=mysql --basedir=/usr/local/mysql --datadir=/data/mysql/3306     初始化mysqld
mysqld --initialize       系统自动生成密码root密码

mysqld --verbose --help   查看mysqld配置

 

5.6初始化

mysql_install_db -u mysql --basedir=/usr/local/mysql --datadir=/data/mysql/3306

 
 
vi /etc/my.cnf
[mysqld]
user=mysql
basedir=/usr/local/mysql
datadir=/data/mysql/3306
port=3306
socket=/tmp/mysql.socket
server_id=6
[mysql]
socket=/tmp/mysql.socket
port=3306
 
 
cp  /usr/local/mysql/support-files/mysql.server /etc/init.d/mysql
service mysql start
