网络服务配置

cd /etc/neutron/plugins/m12

vi m12_conf.ini

type_drivers=flat,vlan    flat 主机和虚拟一个网段

tenant_network_types=flat

mechanism_drivers=linuxbridge   配置网桥驱动

enable_securty_group =true

 

 

vi /etc/neutron/neutron.conf

core_plugin = m12

 

配置网桥走那个网卡

vi  /etc/neutron/plugins/linuxbridge/linuxbridge_conf.ini

network_vlan _ranges=physnet1

physical_interface_mappings =physnet1:eth0    桥接到eth0

enable_securty_group =true
